function switchTab(id) {
  document.querySelectorAll('.tab-btn').forEach((b, i) => {
    b.classList.toggle('active', ['ev','battery','compare'][i] === id);
  });
  document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
  document.getElementById('panel-' + id).classList.add('active');
  if (id === 'ev') calcEV();
  if (id === 'battery') calcBattery();
  if (id === 'compare') calcCompare();
}

/* ---- HELPERS ---- */
const isES = () => document.body.classList.contains('lang-es');

function fmt(n) {
  return '$' + Math.abs(n).toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
}
function fmtI(n) {
  return '$' + Math.round(Math.abs(n)).toLocaleString('en-US');
}

function card(labelEN, labelES, value, cls = '') {
  const label = isES() ? labelES : labelEN;
  const highlight = cls ? 'highlight' : '';
  return `<div class="result-card ${highlight}">
    <div class="result-label">${label}</div>
    <div class="result-value ${cls}">${value}</div>
  </div>`;
}

/* ---- EV CALC ---- */
function calcEV() {
  const [cap, eff] = document.getElementById('ev-model').value.split(',').map(Number);
  const rate = parseFloat(document.getElementById('ev-rate').value) || 0.22;
  const start = parseFloat(document.getElementById('ev-start').value) || 20;
  const target = parseFloat(document.getElementById('ev-target').value) || 80;
  const chargerEff = parseFloat(document.getElementById('ev-charger').value) || 0.88;
  const sessions = parseFloat(document.getElementById('ev-sessions').value) || 20;

  const pct = Math.max(0, target - start) / 100;
  const kwhSession = (cap * pct) / chargerEff;
  const costSession = kwhSession * rate;
  const costMonth = costSession * sessions;
  const costMile = (eff * rate) / chargerEff;
  const costYear = costMonth * 12;

  document.getElementById('ev-results').innerHTML =
    card('Per session', 'Por sesión', fmt(costSession)) +
    card('Per mile', 'Por milla', fmt(costMile)) +
    card('Per month', 'Por mes', fmt(costMonth), 'amber') +
    card('Per year', 'Por año', fmtI(costYear), 'amber');
}

/* ---- BATTERY CALC ---- */
function calcBattery() {
  const cap = parseFloat(document.getElementById('bat-model').value) || 13.5;
  const units = parseFloat(document.getElementById('bat-units').value) || 1;
  const peak = parseFloat(document.getElementById('bat-peak').value) || 0.32;
  const offpeak = parseFloat(document.getElementById('bat-offpeak').value) || 0.18;
  const cycles = parseFloat(document.getElementById('bat-cycles').value) || 1;
  const cost = parseFloat(document.getElementById('bat-cost').value) || 11500;

  const totalCap = cap * units;
  const kwhDay = totalCap * Math.min(cycles, 1) * 0.9;
  const savingsDay = kwhDay * (peak - offpeak);
  const savingsMonth = savingsDay * 30;
  const savingsYear = savingsDay * 365;
  const payback = cost / Math.max(savingsYear, 0.01);

  document.getElementById('bat-results').innerHTML =
    card('Daily savings', 'Ahorro diario', fmt(savingsDay), 'green') +
    card('Monthly savings', 'Ahorro mensual', fmt(savingsMonth), 'green') +
    card('Annual savings', 'Ahorro anual', fmtI(savingsYear), 'green') +
    card('Payback period', 'Período de retorno', payback.toFixed(1) + (isES() ? ' años' : ' yrs'));
}

/* ---- COMPARE CALC ---- */
function calcCompare() {
  const [, eff] = document.getElementById('cmp-ev').value.split(',').map(Number);
  const mpg = parseFloat(document.getElementById('cmp-mpg').value) || 28;
  const kwh = parseFloat(document.getElementById('cmp-kwh').value) || 0.22;
  const gas = parseFloat(document.getElementById('cmp-gas').value) || 3.50;
  const miles = parseFloat(document.getElementById('cmp-miles').value) || 12000;

  const evCostYear = (miles * eff * kwh) / 0.88;
  const gasCostYear = (miles / mpg) * gas;
  const savings = gasCostYear - evCostYear;
  const total = evCostYear + gasCostYear;
  const evPct = Math.round((evCostYear / total) * 100);
  const gasPct = 100 - evPct;

  const evLabel = isES() ? 'VE' : 'EV';
  const gasLabel = isES() ? 'Gasolina' : 'Gas';
  const savingsLabel = isES() ? 'Tu ahorro/año' : 'You save/year';
  const fiveLabel = isES() ? 'Ahorro en 5 años' : '5-yr savings';

  document.getElementById('cmp-bar-section').innerHTML = `
    <div class="vs-section">
      <div class="vs-label">${isES() ? 'Comparación de costo anual de combustible' : 'Annual fuel cost comparison'}</div>
      <div class="vs-bar">
        <div class="ev-seg" style="width:${evPct}%"></div>
        <div class="gas-seg" style="width:${gasPct}%"></div>
      </div>
      <div class="vs-legend">
        <span><span class="dot" style="background:var(--accent)"></span>${evLabel} ${fmtI(evCostYear)}/yr</span>
        <span><span class="dot" style="background:#ef4444"></span>${gasLabel} ${fmtI(gasCostYear)}/yr</span>
      </div>
    </div>`;

  document.getElementById('cmp-results').innerHTML =
    card('EV annual cost', 'Costo anual VE', fmtI(evCostYear), 'green') +
    card('Gas annual cost', 'Costo anual gasolina', fmtI(gasCostYear)) +
    card(savingsLabel, savingsLabel, fmtI(savings), savings > 0 ? 'green' : 'amber') +
    card(fiveLabel, fiveLabel, fmtI(savings * 5), savings > 0 ? 'blue' : 'amber');
}
